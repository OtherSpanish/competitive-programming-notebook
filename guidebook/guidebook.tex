\section{Entrada / Salida Rápida (Fast I/O)}
De más lento (didáctico) a más rápido (para TLE ajustado). Usa el lector por
bytes solo cuando haya del orden de $10^6$–$10^7$ lecturas.

\subsection{C\texttt{++} (g\texttt{++} 8.1, -std=c\texttt{++}17)}
\begin{lstlisting}[style=cpp]
#include <iostream>
using namespace std;
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);           // no mezclar con scanf
    int entero;
    cin >> entero;
    cout << entero << "\n";     // "\n", nunca endl
}
\end{lstlisting}

\subsection{Java}
\begin{lstlisting}[style=java]
BufferedReader br = new BufferedReader(
    new InputStreamReader(System.in));
StringTokenizer st =
    new StringTokenizer(br.readLine());
int entero = Integer.parseInt(st.nextToken());
\end{lstlisting}

\subsection{Python (3.7.4)}
\begin{lstlisting}[style=py]
import sys
def main():
    datos = sys.stdin.buffer.read().split()
    entero = int(datos[0])
    sys.stdout.write(f"{entero}\n")
main()
\end{lstlisting}

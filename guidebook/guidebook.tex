\section{Entrada / Salida Rápida (Fast I/O)}
De más lento (didáctico) a más rápido (para TLE ajustado). Usa el lector por
bytes solo cuando haya del orden de $10^6$–$10^7$ lecturas.

\subsection{C\texttt{++} (g\texttt{++} 8.1, -std=c\texttt{++}17)}
\textbf{Tier 1 — Básico:} \texttt{cin} / \texttt{cout}.
\begin{lstlisting}[style=cpp]
int entero; double decimal; string palabra, linea;
cin >> entero >> decimal >> palabra;  // >> lee UN token
// ! cin >> deja el '\n' pendiente;
// ! el 1er getline sale VACÍO -> hay que ignorarlo
cin.ignore();
getline(cin, linea);                  // línea completa
\end{lstlisting}
\textbf{Tier 2 — Rápido:} desactiva la sync con C (una vez en
\texttt{main}). Punto dulce para casi todo.
\begin{lstlisting}[style=cpp]
ios_base::sync_with_stdio(false);
cin.tie(nullptr);            // ! no mezclar con scanf/printf
cout << entero << "\n";      // "\n", nunca endl (endl frena)
\end{lstlisting}
\textbf{Tier 3 — Más rápido:} lector por bytes con
\texttt{fread}. Solo con $10^6$–$10^7$ lecturas.
\begin{lstlisting}[style=cpp]
struct FastReader {
    static const int TAM = 1 << 16;   // 64 KB
    char bufer[TAM];
    int puntero = 0, leidos = 0;

    int leer() {
        if (puntero == leidos) {
            leidos = (int) fread(bufer, 1, TAM, stdin);
            puntero = 0;
        }
        return leidos == 0 ? -1 : bufer[puntero++]; // -1=EOF
    }
    int nextInt() {
        int resultado = 0, b = leer();
        while (b <= ' ') b = leer();      // saltar blancos
        bool negativo = (b == '-');
        if (negativo) b = leer();
        while (b >= '0' && b <= '9') {
            resultado = resultado * 10 + (b - '0'); b = leer();
        }
        return negativo ? -resultado : resultado;
    }
    long long nextLong() {
        long long resultado = 0; int b = leer();
        while (b <= ' ') b = leer();
        bool negativo = (b == '-');
        if (negativo) b = leer();
        while (b >= '0' && b <= '9') {
            resultado = resultado * 10 + (b - '0'); b = leer();
        }
        return negativo ? -resultado : resultado;
    }
    double nextDouble() {
        double resultado = 0, divisor = 1; int b = leer();
        while (b <= ' ') b = leer();
        bool negativo = (b == '-');
        if (negativo) b = leer();
        while (b >= '0' && b <= '9') {
            resultado = resultado * 10 + (b - '0'); b = leer();
        }
        if (b == '.') {
            b = leer();
            while (b >= '0' && b <= '9') {
                resultado += (b - '0') / (divisor *= 10); b = leer();
            }
        }
        return negativo ? -resultado : resultado;
    }
    string next() {                       // un token
        string s; int b = leer();
        while (b <= ' ') b = leer();
        while (b > ' ') { s += (char) b; b = leer(); }
        return s;
    }
    string nextLine() {                   // línea completa
        string s; int b = leer();
        while (b != '\n' && b != -1) {
            if (b != '\r') s += (char) b; b = leer();
        }
        return s;
    }
};
\end{lstlisting}

\textbf{Tier 4 — Más rápido:} lector por bytes
(\texttt{DataInputStream}). El más veloz sin librerías.
\begin{lstlisting}[style=java]
static class FastReader {
    private final DataInputStream flujo;
    private final byte[] bufer = new byte[1 << 16];
    private int puntero = 0, leidos = 0;

    FastReader() { flujo = new DataInputStream(System.in); }

    int nextInt() throws IOException {
        int resultado = 0, b = leer();
        while (b <= ' ') b = leer();
        boolean negativo = (b == '-');
        if (negativo) b = leer();
        while (b >= '0' && b <= '9') {
            resultado = resultado * 10 + (b - '0'); b = leer();
        }
        return negativo ? -resultado : resultado;
    }
    long nextLong() throws IOException {
        long resultado = 0; int b = leer();
        while (b <= ' ') b = leer();
        boolean negativo = (b == '-');
        if (negativo) b = leer();
        while (b >= '0' && b <= '9') {
            resultado = resultado * 10 + (b - '0'); b = leer();
        }
        return negativo ? -resultado : resultado;
    }
    private int leer() throws IOException {
        if (puntero == leidos) {
            leidos = flujo.read(bufer, 0, bufer.length);
            puntero = 0;
        }
        return leidos == -1 ? -1 : bufer[puntero++];
    }
}
\end{lstlisting}

\textbf{Tier 3 — Más rápido:} leer TODO el \texttt{stdin} de
golpe y tokenizar; la salida, en UNA sola escritura.
\begin{lstlisting}[style=py]
import sys
def main():
    datos = sys.stdin.buffer.read().split()
    indice = 0
    def siguiente_int():      # avanza por los tokens
        nonlocal indice
        valor = int(datos[indice]); indice += 1
        return valor

    n = siguiente_int()
    total = 0
    for _ in range(n):
        total += siguiente_int()
    sys.stdout.write(f"{total}\n")
main()

# Nota: en Python 32-bit ojo con la memoria (~2 GB).

\section{Estructuras de Datos (Data Structures)}

\subsection{Arreglos (Arrays)}
Colección de tamaño \textbf{fijo} del mismo tipo. \textbf{Complejidad:}
acceso/asignación por índice $O(1)$; crear $O(n)$.
\begin{lstlisting}[style=cpp]
int a[5] = {1,2,3,4,5};   // crear: O(n)
a[0] = 10;                // acceso/asignacion: O(1)
\end{lstlisting}
\begin{lstlisting}[style=java]
int[] a = new int[5];     // crear: O(n)
a[0] = 10;                // acceso: O(1)
int n = a.length;         // O(1)  (.length sin ())
\end{lstlisting}
\begin{lstlisting}[style=py]
a = [0] * 5    # crear: O(n)  (no hay arreglo fijo)
a[0] = 10      # acceso: O(1)
\end{lstlisting}

\subsection{Matrices (Arreglos 2D)}
Arreglo bidimensional (filas $\times$ columnas). \textbf{Complejidad:}
acceso \texttt{[i][j]} $O(1)$; crear $n\times m$ en $O(nm)$.
\begin{lstlisting}[style=cpp]
vector<vector<int>> m(3, vector<int>(4,0)); // crear: O(n*m)
m[1][2] = 7;                                 // acceso: O(1)
\end{lstlisting}
\begin{lstlisting}[style=java]
int[][] m = new int[3][4];   // crear: O(n*m)
m[1][2] = 7;                 // acceso: O(1)
\end{lstlisting}
\begin{lstlisting}[style=py]
m = [[0]*4 for _ in range(3)] # crear: O(n*m)
m[1][2] = 7                   # acceso: O(1)
# ! NO [[0]*4]*3 (las filas se comparten)
\end{lstlisting}

\subsection{Vectores / Listas dinámicas}
Arreglo que \textbf{crece} solo (ArrayList / \texttt{vector} /
\texttt{list}). \textbf{Complejidad:} acceso por índice $O(1)$; agregar
al final $O(1)$ amortizado; insertar/borrar en medio $O(n)$; buscar
$O(n)$.
\begin{lstlisting}[style=cpp]
vector<int> v;
v.push_back(3);   // final: O(1) amortizado
int x = v[0];     // acceso: O(1)
int n = v.size(); // O(1)
\end{lstlisting}
\begin{lstlisting}[style=java]
ArrayList<Integer> v = new ArrayList<>();
v.add(3);          // final: O(1) amortizado
int x = v.get(0);  // acceso: O(1)
int n = v.size();  // O(1)
\end{lstlisting}
\begin{lstlisting}[style=py]
v = []
v.append(3)   # final: O(1) amortizado
x = v[0]      # acceso: O(1)
n = len(v)    # O(1)
\end{lstlisting}

\subsection{Listas enlazadas (LinkedList)}
Nodos enlazados. \textbf{Complejidad:} insertar/eliminar en los
\textbf{extremos} $O(1)$; acceso por índice $O(n)$; buscar $O(n)$. Útil
como cola o deque.
\begin{lstlisting}[style=cpp]
list<int> l;              // o deque<int>
l.push_front(1);          // O(1)
l.push_back(9);           // O(1)   acceso [i]: O(n)
\end{lstlisting}
\begin{lstlisting}[style=java]
LinkedList<Integer> l = new LinkedList<>();
l.addFirst(1);            // O(1)
l.addLast(9);             // O(1)
l.removeFirst();          // O(1)   get(i): O(n)
\end{lstlisting}
\begin{lstlisting}[style=py]
from collections import deque
d = deque()
d.appendleft(1)           # O(1)
d.append(9)               # O(1)
d.popleft()               # O(1)   d[i]: O(n)
\end{lstlisting}

\subsection{Conjuntos hash (HashSet)}
Colección \textbf{sin duplicados} y sin orden. \textbf{Complejidad:}
insertar, buscar y borrar $O(1)$ promedio ($O(n)$ peor caso).
\begin{lstlisting}[style=cpp]
unordered_set<int> s;
s.insert(3);            // O(1) prom.
bool hay = s.count(3);  // O(1) prom.
\end{lstlisting}
\begin{lstlisting}[style=java]
HashSet<Integer> s = new HashSet<>();
s.add(3);                     // O(1) prom.
boolean hay = s.contains(3);  // O(1) prom.
\end{lstlisting}
\begin{lstlisting}[style=py]
s = set()
s.add(3)      # O(1) prom.
hay = 3 in s  # O(1) prom.
\end{lstlisting}

\subsection{Mapas / Diccionarios hash}
Pares \textbf{clave $\rightarrow$ valor} sin orden (HashMap /
\texttt{unordered\_map} / \texttt{dict}). \textbf{Complejidad:}
insertar, acceder y borrar por clave $O(1)$ promedio ($O(n)$ peor
caso).
\begin{lstlisting}[style=cpp]
unordered_map<string,int> m;
m["a"] = 1;      // insertar/asignar: O(1) prom.
int v = m["a"];  // acceso: O(1) prom.
\end{lstlisting}
\begin{lstlisting}[style=java]
HashMap<String,Integer> m = new HashMap<>();
m.put("a", 1);                    // O(1) prom.
int v = m.getOrDefault("a", 0);   // O(1) prom.
\end{lstlisting}
\begin{lstlisting}[style=py]
m = {}
m["a"] = 1          # O(1) prom.
v = m.get("a", 0)   # O(1) prom.  (get con default)
\end{lstlisting}

\subsection{Conjuntos ordenados (TreeSet)}
Conjunto sin duplicados \textbf{mantenido ordenado} (árbol balanceado).
\textbf{Complejidad:} insertar, buscar, borrar y piso/techo
$O(\log n)$.
\begin{lstlisting}[style=cpp]
set<int> s;
s.insert(5); s.insert(1);    // O(log n) c/u
int menor = *s.begin();      // min: O(1)
auto it = s.lower_bound(3);  // techo: O(log n)
\end{lstlisting}
\begin{lstlisting}[style=java]
TreeSet<Integer> s = new TreeSet<>();
s.add(5); s.add(1);          // O(log n) c/u
int menor = s.first();       // O(log n)
Integer techo = s.ceiling(3);// O(log n)
\end{lstlisting}
\begin{lstlisting}[style=py]
# ! Python NO trae TreeSet built-in
from sortedcontainers import SortedSet
s = SortedSet()
s.add(5); s.add(1)     # O(log n) c/u
menor = s[0]           # min: O(log n)
i = s.bisect_left(3)   # techo: O(log n)
\end{lstlisting}

\subsection{Mapas ordenados (TreeMap)}
Mapa clave $\rightarrow$ valor \textbf{ordenado por clave} (árbol
balanceado). \textbf{Complejidad:} insertar, acceder, borrar y
floor/ceiling $O(\log n)$.
\begin{lstlisting}[style=cpp]
map<string,int> m;
m["b"] = 2;  m["a"] = 1;     // O(log n) c/u
auto primera = m.begin();    // clave menor: O(1)
\end{lstlisting}
\begin{lstlisting}[style=java]
TreeMap<String,Integer> m = new TreeMap<>();
m.put("b", 2);  m.put("a", 1);  // O(log n) c/u
String prim = m.firstKey();     // O(log n)
\end{lstlisting}
\begin{lstlisting}[style=py]
# ! dict NO ordena por clave (conserva insercion)
from sortedcontainers import SortedDict
m = SortedDict()
m["b"] = 2; m["a"] = 1   # O(log n) c/u
prim = m.peekitem(0)     # menor clave: O(log n)
\end{lstlisting}

\subsection{Pilas (Stack)}
Estructura \textbf{LIFO} (último en entrar, primero en salir).
\textbf{Complejidad:} push, pop y consultar el tope $O(1)$. Para DFS
iterativo, paréntesis balanceados, deshacer.
\begin{lstlisting}[style=cpp]
stack<int> pila;
pila.push(3);          // O(1)
int tope = pila.top(); // O(1)
pila.pop();            // O(1)
\end{lstlisting}
\begin{lstlisting}[style=java]
Deque<Integer> pila = new ArrayDeque<>(); // no uses Stack
pila.push(3);              // O(1)
int tope = pila.peek();    // O(1)
pila.pop();                // O(1)
\end{lstlisting}
\begin{lstlisting}[style=py]
pila = []
pila.append(3)   # push: O(1)
tope = pila[-1]  # O(1)
pila.pop()       # O(1)
\end{lstlisting}

\subsection{Colas (Queue)}
Estructura \textbf{FIFO} (primero en entrar, primero en salir).
\textbf{Complejidad:} encolar y desencolar $O(1)$. Base del BFS.
\begin{lstlisting}[style=cpp]
queue<int> cola;
cola.push(3);              // encolar: O(1)
int frente = cola.front(); // O(1)
cola.pop();                // desencolar: O(1)
\end{lstlisting}
\begin{lstlisting}[style=java]
Queue<Integer> cola = new ArrayDeque<>();
cola.offer(3);             // encolar: O(1)
int frente = cola.peek();  // O(1)
cola.poll();               // desencolar: O(1)
\end{lstlisting}
\begin{lstlisting}[style=py]
from collections import deque
cola = deque()
cola.append(3)   # encolar: O(1)
frente = cola[0] # O(1)
cola.popleft()   # desencolar: O(1)
\end{lstlisting}

\subsection{Colas de prioridad (Priority Queue / Heap)}
Devuelve siempre el elemento de mayor (o menor) prioridad.
\textbf{Complejidad:} insertar y extraer $O(\log n)$; consultar el tope
$O(1)$. Para Dijkstra, Prim, k-ésimo mayor.
\begin{lstlisting}[style=cpp]
priority_queue<int> pq;    // MAX-heap por defecto
pq.push(3);          // O(log n)
int mx = pq.top();   // O(1)  (el mayor)
pq.pop();            // O(log n)
// min-heap:
priority_queue<int, vector<int>, greater<int>> pqMin;
\end{lstlisting}
\begin{lstlisting}[style=java]
PriorityQueue<Integer> pq = new PriorityQueue<>(); // MIN-heap
pq.offer(3);         // O(log n)
int mn = pq.peek();  // O(1)  (el menor)
pq.poll();           // O(log n)
// max-heap: new PriorityQueue<>(Collections.reverseOrder())
\end{lstlisting}
\begin{lstlisting}[style=py]
import heapq
pq = []
heapq.heappush(pq, 3)  # O(log n)
mn = pq[0]             # O(1)  (el menor; heapq es MIN-heap)
heapq.heappop(pq)      # O(log n)
# max-heap: guarda los valores negados (-x)
\end{lstlisting}

\section{Búsqueda Binaria (Binary Search)}
Requiere datos \textbf{ordenados} o un \textbf{predicado monótono}
(pasa de falso a verdadero una sola vez). \textbf{Complejidad:}
$O(\log n)$ en discreto; $O(\text{iter})$ en reales;
$O(L\log n)$ con cadenas. Tip: usa \texttt{lo+(hi-lo)/2} para el
\texttt{mid} y evitar overflow.

\subsection{En arreglo ordenado}
Primer índice $\geq x$ (\texttt{lower\_bound}) o $> x$
(\texttt{upper\_bound}).
\begin{lstlisting}[style=cpp]
int i = lower_bound(a.begin(),a.end(),x)-a.begin(); // >= x
int j = upper_bound(a.begin(),a.end(),x)-a.begin(); // >  x
bool hay = binary_search(a.begin(),a.end(),x);
\end{lstlisting}
\begin{lstlisting}[style=java]
int idx = Arrays.binarySearch(a, x); // >=0 si existe
// lower_bound manual:
int lo=0, hi=a.length;
while (lo<hi){ int m=(lo+hi)>>>1;
  if(a[m]<x) lo=m+1; else hi=m; }  // lo = primer >= x
\end{lstlisting}
\begin{lstlisting}[style=py]
from bisect import bisect_left, bisect_right
i = bisect_left(a, x)    # primer >= x
j = bisect_right(a, x)   # primer >  x
\end{lstlisting}

\subsection{Menor $x$ que cumple}
Predicado monótono F..F\,T..T; busca la \textbf{frontera} izquierda.
\begin{lstlisting}[style=cpp]
long long res=-1;
while(lo<=hi){ long long mid=lo+(hi-lo)/2;
  if(cumple(mid)){res=mid; hi=mid-1;} // sirve -> izq
  else lo=mid+1; }
\end{lstlisting}
\begin{lstlisting}[style=java]
long res=-1;
while(lo<=hi){ long mid=lo+(hi-lo)/2;
  if(cumple(mid)){res=mid; hi=mid-1;}
  else lo=mid+1; }
\end{lstlisting}
\begin{lstlisting}[style=py]
res = -1
while lo <= hi:
    mid = (lo + hi) // 2
    if cumple(mid): res = mid; hi = mid - 1
    else: lo = mid + 1
\end{lstlisting}

\subsection{Mayor $x$ que cumple}
Predicado monótono T..T\,F..F; busca la \textbf{frontera} derecha
(solo cambian dos líneas respecto al anterior).
\begin{lstlisting}[style=cpp]
long long res=-1;
while(lo<=hi){ long long mid=lo+(hi-lo)/2;
  if(cumple(mid)){res=mid; lo=mid+1;} // sirve -> der
  else hi=mid-1; }
\end{lstlisting}
\begin{lstlisting}[style=java]
long res=-1;
while(lo<=hi){ long mid=lo+(hi-lo)/2;
  if(cumple(mid)){res=mid; lo=mid+1;}
  else hi=mid-1; }
\end{lstlisting}
\begin{lstlisting}[style=py]
res = -1
while lo <= hi:
    mid = (lo + hi) // 2
    if cumple(mid): res = mid; lo = mid + 1
    else: hi = mid - 1
\end{lstlisting}

\subsection{Sobre reales}
Respuesta continua (raíces, puntos de equilibrio); iteras un número
fijo de veces en vez de comparar $lo\leq hi$.
\begin{lstlisting}[style=cpp]
for(int it=0; it<100; it++){ double mid=(lo+hi)/2;
  if(cumple(mid)) hi=mid; else lo=mid; }
// lo ~ hi = frontera
\end{lstlisting}
\begin{lstlisting}[style=java]
for(int it=0; it<100; it++){ double mid=(lo+hi)/2;
  if(cumple(mid)) hi=mid; else lo=mid; }
\end{lstlisting}
\begin{lstlisting}[style=py]
for _ in range(100):
    mid = (lo + hi) / 2
    if cumple(mid): hi = mid
    else: lo = mid
\end{lstlisting}

\subsection{Con strings}
Igual que en arreglo ordenado, pero comparación lexicográfica: cada
comparación cuesta $O(L)$, total $O(L\log n)$.
\begin{lstlisting}[style=cpp]
// vector<string> s ordenado
int i = lower_bound(s.begin(),s.end(),string("caro"))
        - s.begin();
bool hay = binary_search(s.begin(),s.end(),string("caro"));
\end{lstlisting}
\begin{lstlisting}[style=java]
int idx = Arrays.binarySearch(s, "caro"); // usa compareTo
// manual: if(s[m].compareTo(x) < 0) lo=m+1; else hi=m;
\end{lstlisting}
\begin{lstlisting}[style=py]
i = bisect_left(s, "caro")   # str comparan lexicografico
hay = i < len(s) and s[i] == "caro"
\end{lstlisting}
